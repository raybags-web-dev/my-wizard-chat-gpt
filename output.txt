function getKeys(obj) {
  let keys = [];
  for (let key in obj) {
    keys.push(key); 
  }
  return keys;
}

Example: 
let students = {
  student1: 'John',
  student2: 'Alice',
  student3: 'Bob'
};

let studentKeys = getKeys(students);
console.log(studentKeys); // Output: ['student1', 'student2', 'student3']
